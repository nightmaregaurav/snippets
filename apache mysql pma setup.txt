install mariadb/mysql apache phpmyadmin server using package manager

sudo mysql_install_db --user=mysql --basedir=/usr --datadir=/var/lib/mysql
systemctl restart mysqld
sudo mysql_upgrade
sudo mysql -u root -p

CREATE USER 'pma'@'localhost' identified by 'pma';
grant all on *.* to 'pma'@'localhost';
commit;
exit;


Create Script: amsqstart => pkexec sudo sh -c 'systemctl start mysqld && apachectl start'
Create Script: amsqstop => pkexec sudo sh -c 'systemctl stop mysqld && apachectl stop'
Create Script: amsqrestart => pkexec sudo sh -c 'systemctl restart mysqld && apachectl restart'
