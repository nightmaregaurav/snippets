install mariadb mariadb-client using package manager

sudo mysql_install_db --user=mysql --basedir=/usr --datadir=/var/lib/mysql
systemctl enable --now mariadb.service
systemctl start mariadb.service
sudo mysql_upgrade --force
sudo mysql_secure_installation  
sudo mysql -u root -p
CREATE USER 'pma'@'localhost' identified by 'pma';
grant all on *.* to 'pma'@'localhost';
FLUSH PRIVILEGES;
ALTER USER 'root'@'localhost' IDENTIFIED VIA mysql_native_password USING PASSWORD('');
commit;
exit;

install apache php php-apache from package manager 

In /etc/httpd/conf/httpd.conf, comment the line:
#LoadModule mpm_event_module modules/mod_mpm_event.so

and uncomment the line:
LoadModule mpm_prefork_module modules/mod_mpm_prefork.so

Place this at the end of the LoadModule list:
LoadModule php_module modules/libphp.so
AddHandler php-script .php

Place this at the end of the Include list:
# php configuration
Include conf/extra/php_module.conf

systemctl enable --now httpd.service
systemctl start httpd.service

install phpmyadmin from package manager

Create File /etc/httpd/conf/extra/phpmyadmin.conf and write:
Alias /phpmyadmin "/usr/share/webapps/phpMyAdmin"
<Directory "/usr/share/webapps/phpMyAdmin">
    AllowOverride All
    Options FollowSymlinks
    Require all granted
</Directory>

Edit /etc/httpd/conf/httpd.conf and write at last:
# phpMyAdmin configuration
Include conf/extra/phpmyadmin.conf

mkdir -p /var/lib/phpmyadmin/tmp
chown -R http:http /var/lib/phpmyadmin/tmp
mkdir /etc/phpmyadmin
cp /usr/share/webapps/phpMyAdmin/config.sample.inc.php /usr/share/webapps/phpMyAdmin/config.inc.php

Edit /usr/share/webapps/phpMyAdmin/config.inc.php and make these changes:
$cfg["TempDir"] = "/var/lib/phpmyadmin/tmp";
$cfg[‘blowfish_secret’] = "replace with 32bit randomstrings";

Edit /etc/php/php.ini file and uncomment the following:
extension=mysqli
extension=curl
extension=iconv

systemctl restart httpd
